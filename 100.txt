import jwt 
import uuid
import hashlib

import requests

secretKey = '발급받은 SECRET KEY'

}

headers = {
  'Authorization': authorization_token
}

try:
    # Call API
    response = requests.get(apiUrl + '/v1/orders/chance', params=param, headers=headers)
    # handle to success or fail

    print(response.json())
except Exception as err:
    # handle exception
    print(err)

입력 규칙

# Python 3
  # pip3 installl pyJwt
  import jwt
  import uuid
  import hashlib
  import time
  from urllib.parse import urlencode
  import requests

  accessKey = '발급받은 API KEY'
  secretKey = '발급받은 SECRET KEY'
  apiUrl = 'https://api.bithumb.com'

  # Set API parameters
  param = dict( market='KRW-XRP', limit=100, page=1, order_by='desc' )

  # uuids 또는 client_order_ids 두 값 모두 전달하지 않은 경우 최신 주문 내역 조회
  # 두 값이 모두 전달된 경우, uuids를 기준으로 조회
  uuids = [
      'C0106000032400700021', 'C0106000043000097801'
  ]
  client_order_ids = [
      'my-order-001', 'my-order-002'
  ]

  query = urlencode(param)
  uuid_query = '&'.join([f'uuids[]={u}' for u in uuids])
  client_order_id_query = '&'.join([f'client_order_ids[]={id}' for id in client_order_ids])

  if uuid_query:
      query = query + "&" + uuid_query
  if client_order_id_query:
      query = query + "&" + client_order_id_query

  # Generate access token
  hash = hashlib.sha512()
  hash.update(query.encode())
  query_hash = hash.hexdigest()
  payload = {
      'access_key': accessKey,
      'nonce': str(uuid.uuid4()),
      'timestamp': round(time.time() * 1000),
      'query_hash': query_hash,
      'query_hash_alg': 'SHA512',
  }
  jwt_token = jwt.encode(payload, secretKey)
  authorization_token = 'Bearer {}'.format(jwt_token)
  headers = {
    'Authorization': authorization_token
  }

  try:
      # Call API
      response = requests.get(apiUrl + '/v1/orders?' + query, headers=headers)
      # handle to success or fail
      print(response.status_code)
      print(response.json())
  except Exception as err:
      # handle exception
      print(err)

다건 주문 요청
post
https://api.bithumb.com/v2/orders/batch

여러 건의 주문을 한 번의 요청으로 생성합니다. 요청당 최대 20건까지 가능합니다.

예시코드는 JavaScript, Python, JAVA에 한해서만 제공합니다.

📌
입력 규칙
지정가 주문: price와 volume 모두 필요
시장가 매수: price(총 금액) 필요
시장가 매도: volume 필요
client_order_id 는 회원 단위로 고유해야 합니다. 동일 요청(batch_orders) 내에서 중복된 ID가 포함된 경우, 최초 항목만 정상 접수되며 이후 중복 건은 실패 처리됩니다.
📘
안내 사항
서버가 요청을 정상적으로 수신 및 처리한 경우, 개별 주문 중 일부가 실패하더라도 상태 코드는 200 OK를 반환합니다.

현재 API 호출 제한은 IP별 초당 20회로 설정되어 있습니다. 다만, 시스템 상황이나 트래픽에 따라 이 제한은 변경될 수 있습니다.

기존에는 자전거래 가능성이 있을 경우 주문이 접수되지 않았으나, 개선 후에는 주문 접수 후 체결 단계에서 자전거래를 감지해 충돌하는 신규 주문 수량만 취소하도록 개선될 예정입니다. 자세한 내용은 공지사항을 참고하세요(업데이트 일시: 2026년 4월 2일 오후 12시 이후).

응답 항목 "자전거래 방지 유형" stp_type이 신규 추가될 예정입니다.

Request
  # Python 3
  # pip3 installl pyJwt
  import jwt
  import uuid
  import hashlib
  import time
  import requests
  import json

  accessKey = '발급받은 API KEY'
  secretKey = '발급받은 SECRET KEY'
  apiUrl = 'https://api.bithumb.com'

  # Set API parameters - 다건 주문 (최대 20건)
  batch_orders = [
      dict(market='KRW-BTC', side='bid', order_type='limit', price=84000000, volume=0.001, client_order_id='my-order-001'),
      dict(market='KRW-BTC', side='bid', order_type='limit', price=83000000, volume=0.001)
  ]

  requestBody = dict(batch_orders=batch_orders)

  # Generate query string for hash (배열 형식 처리)
  query_parts = []
  for index, order in enumerate(batch_orders):
      for key, value in order.items():
          query_parts.append(f'batch_orders[{index}][{key}]={value}')
  query = '&'.join(query_parts)

  # Generate access token
  hash = hashlib.sha512()
  hash.update(query.encode())
  query_hash = hash.hexdigest()
  payload = {
      'access_key': accessKey,
      'nonce': str(uuid.uuid4()),
      'timestamp': round(time.time() * 1000),
      'query_hash': query_hash,
      'query_hash_alg': 'SHA512',
  }
  jwt_token = jwt.encode(payload, secretKey)
  authorization_token = 'Bearer {}'.format(jwt_token)
  headers = {
    'Authorization': authorization_token,
    'Content-Type': 'application/json'
  }

  try:
      # Call API
      response = requests.post(apiUrl + '/v2/orders/batch', data=json.dumps(requestBody), headers=headers)
      # handle to success or fail
      print(response.status_code)
      print(response.json())
  except Exception as err:
      # handle exception
      print(err)

다건 주문 취소 접수
post
https://api.bithumb.com/v2/orders/cancel

여러 개의 주문을 일괄 취소 요청합니다. 요청당 최대 30건까지 처리할 수 있습니다.

예시코드는 JavaScript, Python, JAVA에 한해서만 제공합니다.

📌
입력 규칙
order_ids 또는 client_order_ids 중 하나 이상을 반드시 전달해야 하며, 두 값을 모두 전달한 경우 order_ids를 기준으로 처리합니다.

📘
안내사항
현재 API 호출 제한은 IP별 초당 20회로 설정되어 있습니다. 다만, 시스템 상황이나 트래픽에 따라 이 제한은 변경될 수 있습니다.

Body Params
order_ids
array of strings
취소할 주문의 고유 ID(구 uuid) 목록(max 30)

client_order_ids
array of strings
서버에서 부여하는 주문 ID(구 uuid)와 별도로 주문 생성 시 사용자가 직접 지정한 고유 ID 목록(max 30)

허용 문자: 영문 대/소문자, 숫자, -, _
길이: 1–36자
Headers
Authorization
string
required
Authorization token (JWT)

  # Python 3
  # pip3 install pyJwt
  import jwt
  import uuid

  import time
